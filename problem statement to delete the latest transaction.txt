#include <stdio.h>
#include <stdlib.h>

struct transaction {
    int amount;
    struct transaction *next;
};

void deleteFirst(struct transaction **head) {
    struct transaction *temp;

    if (*head == NULL) {
        printf("No transactions\n");
        return;
    }

    temp = *head;
    *head = (*head)->next;
    free(temp);
}

void display(struct transaction *head) {
    while (head != NULL) {
        printf("%d->", head->amount);
        head = head->next;
    }
    printf("NULL\n");
}

int main() {
    struct transaction *head = NULL;

    // Inpu
    head = malloc(sizeof(struct transaction));
    head->amount = 500;

    head->next = malloc(sizeof(struct transaction));
    head->next->amount = 1000;
    head->next->next = NULL;

    // Delete old transaction
    deleteFirst(&head);

    display(head);

    return 0;
}
