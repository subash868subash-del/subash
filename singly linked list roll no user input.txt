#include<stdio.h>
#include<stdlib.h>

struct Rollno{
    long int number;
    struct Rollno *next;//nested structure
};
int main(){
    //storage allocation
    struct Rollno *pr1=(struct Rollno*) malloc(sizeof(struct Rollno));
    struct Rollno *pr2=(struct Rollno*) malloc(sizeof(struct Rollno));

    //gettin gfrom  the user
    printf("Enter the first Rollnumber:\n");
    scanf("%d",&pr1->number);
    
    printf("Enter the second ROllnumber:\n");
    scanf("%d",&pr2->number);
    

    //interconnection
    pr1->next = pr2;
    pr2->next = NULL;
 
    //POINTING head
    struct Rollno *head =pr1;
    //display
    printf("Rollno: \n");
    while(head != NULL){
    printf("Roll no %ld --->", head->number);
    head = head->next;
}
   printf("NULL");

    return 0;
}
