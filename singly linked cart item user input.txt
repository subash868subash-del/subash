#include<stdio.h>
#include<stdlib.h>

struct Item{
    int number;
    struct Item  *next;//nested structure
};
int main(){
    //storage allocation
    struct Item *pr1=(struct Item*) malloc(sizeof(struct Item));
    struct Item *pr2=(struct Item*) malloc(sizeof(struct Item));
    struct Item *pr3=(struct Item*) malloc(sizeof(struct Item));
    //gettin gfrom  the user
    printf("Enter the first Item:\n");
    scanf("%d",&pr1->number);
    
    printf("Enter the second Item:\n");
    scanf("%d",&pr2->number);
    printf("Enter the third Item:\n");
    scanf("%d",&pr3->number);

    //interconnection
    pr1->next = pr2;
    pr2->next = pr3;
    pr3->next = NULL; 
    //POINTING head
    struct Item *head =pr1;
    //display
    printf("ITEMS: \n");
    while(head != NULL){
    printf(" %ld --->", head->number);
    head = head->next;
}
   printf("NULL");

    return 0;
}
